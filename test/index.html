let web = io.web()

fun handle(req: Request) {
    if req.path == "/" {
        return web.page(io.read("index.html"))
    } else if req.path == "/api/data" {
        return web.json("{\"message\": \"Hello!\"}")
    } else {
        return web.error.text(404, "Not found!")
    }
}

io.listen(8080, handle)
